Group by
1.Contare quanti iscritti ci sono stati ogni anno

SELECT COUNT(id) AS `nr_studenti`, YEAR(`enrolment_date`) AS `inizio`
FROM `students`
WHERE `enrolment_date`>= 2018
GROUP BY YEAR(`enrolment_date`)
